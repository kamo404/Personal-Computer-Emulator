.data num1 6
.data num2 4
.data result 0
.data one 1

MAIN:
    CALL MULTIPLY
    LOAD_ACC result
    OUT
    HALT

MULTIPLY:
    LOAD_ACC num2
    JMP_IZ DONE
    LOAD_R1 result
    LOAD_R2 num1
    ADD
    STORE_ACC result
    LOAD_R1 num2
    LOAD_R2 one
    SUB
    STORE_ACC num2
    JMP MULTIPLY
DONE:
    RETURN
